       - Constructor takes `IHttpClientFactory httpFactory`
       - `SubscribeSession(string sessionId, CancellationToken ct)` -> `ChannelReader<SessionEvent>`:
         Creates unbounded Channel, spawns background Task.Run that:
         - Creates HttpClient from factory ("SseClient" named client)
         - GET `/api/sessions/{sessionId}/events` with Accept: text/event-stream

       - `SubscribeFleet(CancellationToken ct)` -> `ChannelReader<SessionEvent>`:

    4. Create `src/AgentHub.Web/Program.cs`:
       ```csharp
       using MudBlazor.Services;
       using AgentHub.Web.Services;

       var builder = WebApplication.CreateBuilder(args);
       builder.Services.AddRazorComponents().AddInteractiveServerComponents();
       builder.Services.AddMudServices();

       builder.Services.AddHttpClient<DashboardApiClient>(client =>
       {
           client.BaseAddress = new Uri("https+http://agenthub-service");
       });
       builder.Services.AddHttpClient("SseClient", client =>
       {
           client.BaseAddress = new Uri("https+http://agenthub-service");
           client.Timeout = Timeout.InfiniteTimeSpan;
       });
       builder.Services.AddScoped<SseStreamService>();

       var app = builder.Build();
       if (!app.Environment.IsDevelopment()) { app.UseHsts(); }
       app.UseStaticFiles();
       app.UseAntiforgery();
       app.MapRazorComponents<AgentHub.Web.Components.App>()
           .AddInteractiveServerRenderMode();
       app.Run();
       ```
