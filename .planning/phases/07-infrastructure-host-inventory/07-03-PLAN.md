    2. In HostSidebar.razor, add inventory expansion support:
       - Add `@inject DashboardApiClient Api` at top (or use cascading parameter if DashboardApiClient is already available via parent)
       - Add to @code block:
         ```csharp
         private string? ExpandedHostId { get; set; }
         private bool _refreshing;

         private void ToggleExpand(string hostId)
         {
             ExpandedHostId = ExpandedHostId == hostId ? null : hostId;
         }

         private async Task RefreshInventory(string hostId)
         {
             _refreshing = true;
             StateHasChanged();
             try { await Api.RefreshInventoryAsync(hostId); }
             finally { _refreshing = false; StateHasChanged(); }
         }
         ```
       - After the MEM bar section (after the closing else block at ~line 92), inside the MudCardContent, add an expand toggle icon and MudCollapse:
         ```razor
         <MudStack Row AlignItems="AlignItems.Center" Class="mt-1">
             <MudSpacer />
             <MudIconButton Icon="@(ExpandedHostId == host.HostId ? Icons.Material.Filled.ExpandLess : Icons.Material.Filled.ExpandMore)"
                            Size="Size.Small"
                            OnClick="() => ToggleExpand(host.HostId)" />
         </MudStack>

         <MudCollapse Expanded="@(ExpandedHostId == host.HostId)">
             <MudDivider Class="my-1" />
             <MudText Typo="Typo.caption" Class="mb-1"><b>Installed Agents</b></MudText>
             @if (host.Inventory?.Agents is { Count: > 0 })
             {
                 @foreach (var agent in host.Inventory.Agents)
                 {
                     <MudText Typo="Typo.caption">@agent.Name @(agent.Version ?? "?")</MudText>
                 }
             }
             else
             {
                 <MudText Typo="Typo.caption" Color="Color.Tertiary">Not probed</MudText>
             }

             <MudText Typo="Typo.caption" Class="mt-1">
                 <b>Disk:</b> @(host.Inventory?.DiskFreeGb.HasValue == true ? $"{host.Inventory.DiskFreeGb:F1} GB free" : "--")
             </MudText>

             @if (host.Inventory?.GitVersion is not null)
             {
                 <MudText Typo="Typo.caption">
                     <b>Git:</b> @host.Inventory.GitVersion
                 </MudText>
             }

             <MudIconButton Icon="@Icons.Material.Filled.Refresh"
                            Size="Size.Small"
                            Disabled="@_refreshing"
                            OnClick="() => RefreshInventory(host.HostId)"
                            Class="mt-1" />
         </MudCollapse>
         ```
